generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

//
// ---------- USERS ----------
//
model User {
  id           Int           @id                      
  password     String
  firstName    String
  lastName     String
  displayName  String
  dept         String
  createdAt    DateTime      @default(now())
  updateSource String        @default("WEB")
  email        String?
  isDeleted    Boolean?      @default(false)

  groups       GroupMember[] //  
}

//
// ---------- GROUPS ----------
//
model Group {
  id          Int           @id @default(autoincrement())
  name        String
  createdAt   DateTime      @default(now())
  createdBy   String
  members     GroupMember[]
  is_deleted  Boolean?      @default(false)          
  email       String?
}

//
// ---------- GROUP MEMBERS ----------
//
model GroupMember {
  id       Int   @id @default(autoincrement())
  userId   Int
  groupId  Int
  group    Group @relation(fields: [groupId], references: [id])
  user     User  @relation(fields: [userId], references: [id])

  @@unique([userId, groupId])  
}

 

    model Template {
      id          Int        @id @default(autoincrement())
      name        String
      description String
      createdBy   String
      createdOn   DateTime   @default(now())
      label       String
      isDeleted   Boolean?   @default(false)

      subtasks    SubTask[]
      tasks       Task[]   
    }


 
model SubTask {
  id           Int       @id @default(autoincrement())
  action       String
  dependsOn    String?
  description  String
  assignee     String
  templateId   Int
  template     Template  @relation(fields: [templateId], references: [id])
  
}

model Task {
  id         Int       @id @default(autoincrement())
  name       String 
  templateId Int
  createdBy  String
  createdOn  DateTime  @default(now())
  isDeleted  Boolean   @default(false)  
  template   Template  @relation(fields: [templateId], references: [id])
}
